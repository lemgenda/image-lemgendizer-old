import { Conv2D as _ag_Conv2D, util as _ag_util } from '@tensorflow/tfjs-core';
// No import for conv2DImpl needed as we delegate to originalConv2d

// [Antigravity] Global Stats Init
if (!globalThis._antigravity) globalThis._antigravity = {};
if (!globalThis._antigravity.conv2d) globalThis._antigravity.conv2d = { count: 0, cpuTime: 0 };

export function conv2d(args) {
    const t0 = performance.now();
    const { inputs, backend, attrs } = args;
    const { x, filter } = inputs;
    const { strides, pad, dataFormat, dilations, dimRoundingMode } = attrs;

    // Delegate to originalConv2d (renamed by postinstall logic)
    const res = originalConv2d(args);

    const dt = performance.now() - t0;
    if (globalThis._antigravity.conv2d) {
        globalThis._antigravity.conv2d.count++;
        globalThis._antigravity.conv2d.cpuTime += dt;
    }
    return res;
}

export const conv2DConfig = {
    kernelName: _ag_Conv2D,
    backendName: 'webgpu',
    kernelFunc: conv2d
};
